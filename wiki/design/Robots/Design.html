import time
import math

class WitherStabilizationController:
    def __init__(self, target_roll=0.0, target_pitch=0.0):
        self.Kp_gimbal = 1.25
        self.Kd_gimbal = 0.45
        self.prev_roll_err = 0.0
        self.prev_pitch_err = 0.0
        self.last_timestamp = time.time()

    def calculate_gimbal_actuation(self, imu_roll, imu_pitch, imu_gyro_x, imu_gyro_y):
        """
        Dynamically adjusts the 2-axis gimbal tilt angles to offset the pendulum
        destabilization effect. Symmetrical dual arm motors compensate for thrust vectors.
        """
        now = time.time()
        dt = max(now - self.last_timestamp, 0.001)
        self.last_timestamp = now

        # Compute error states against horizontal axis
        roll_err = 0.0 - imu_roll
        pitch_err = 0.0 - imu_pitch

        # Gimbal PID servo correction
        roll_correction = (roll_err * self.Kp_gimbal) + (imu_gyro_x * self.Kd_gimbal)
        pitch_correction = (pitch_err * self.Kp_gimbal) + (imu_gyro_y * self.Kd_gimbal)

        # Map to physical servo angles (within hardware bounds +/- 35 degrees)
        left_servo_deg = max(min(1500 + int(roll_correction * 10), 1850), 1150)
        right_servo_deg = max(min(1500 - int(roll_correction * 10), 1850), 1150)

        return left_servo_deg, right_servo_deg

import time

class WardenEcholocationArray:
    def __init__(self, mic_spacing_meters=0.15):
        self.sound_speed_mps = 343.0
        self.mic_spacing = mic_spacing_meters

    def process_ultrasonic_echo(self, left_mic_arrival, right_mic_arrival):
        """
        Performs Time-Difference-Of-Arrival (TDOA) calculations over the MEMS array
        to determine distance and precise angle of obstacles ahead.
        """
        # Time difference in seconds
        tdoa = abs(left_mic_arrival - right_mic_arrival)

        if tdoa > (self.mic_spacing / self.sound_speed_mps):
            # Physical impossibility filter due to noise spikes
            return None, None

        # Triangulate heading angle (theta) from perpendicular plane
        angle_rad = math.asin((self.sound_speed_mps * tdoa) / self.mic_spacing)
        heading_deg = math.degrees(angle_rad)

        # Average distance estimation
        avg_time = (left_mic_arrival + right_mic_arrival) / 2.0
        distance_meters = (avg_time * self.sound_speed_mps) / 2.0

        return distance_meters, heading_deg

Operational Safety & Flight Standards

Mandatory fleet thresholds for real-world deployment safety

1. The 15-Minute Continuous Flight Metric

Under simulated gravitational loads, flying agents must achieve a minimum of 15 minutes of station-keeping (hover) with at least 15% battery potential remaining. Peak motor efficiency must occur near 50% throttle to guarantee structural safety margins in updrafts.

2. Automated "Perch Fail-Safe"

Referencing safety policies modeled directly in the **Ender_Dragon_Sanctuary**, any airborne vehicle must auto-detect solid elevated surfaces. When power declines below 10%, flight coordination must prioritize immediate soft landing or perching.
